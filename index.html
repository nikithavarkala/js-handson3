Find the number of digits

    {
        let c=0;
        while(N!==0){
            c++;
            N=Math.floor(N/10);
        }
        return c;
    };

Find the Fives.

    const Find_Five = (N) => 
    {
        let c=0;
        while(N>0){
            if(N%10===5){
                c++;
            }
            N=Math.floor(N/10);
        }
        return c;
    };

Find Sum

    const findSum = (n) => {
        let sum=0;
        for(let i=1;i<=n;i++){
            if(i%2===0){
                sum+=i;
            }
        }
        return sum;
    };

Find the sum of the digits of a given number.

    const Number_Sum = (N) => 
    {
        let sum=0;
        while(N>0){
            r=N%10;
            sum+=r;
            N=Math.floor(N/10);
        }
        return sum;
    };

Check whether a Number is a prime or not.

    const Prime_Check = (N) => 
    {
        let c=0;
        for(let i=1;i<=N;i++){
            if(N%i===0){
                c++;
            }
        }
        if(c===2){
            return "YES";
        }
        else{
            return "NO";
        }
    };
